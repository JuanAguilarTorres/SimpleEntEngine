401bool LeftCollision(Entity e, Entity other) {
402 const auto& eCollider = e.GetComponent<BoxColliderComponent>();
403 const auto& eTransform = e.GetComponent<TransformComponent>();
404
405 const auto& oCollider = other.GetComponent<BoxColliderComponent>();
406 const auto& oTransform = other.GetComponent<TransformComponent>();
407
408 float eX = eTransform.previousPosition.x;
409 float eY = eTransform.previousPosition.y;
410 float eH = static_cast<float>(eCollider.height);
411
412 float oX = oTransform.previousPosition.x;
413 float oY = oTransform.previousPosition.y;
414 float oH = static_cast<float>(oCollider.height);
415
416 // Left side collides with Other
417 return (
418 oY < eY + eH &&
419 oY + oH > eY &&
420 oX < eX
421 );
422}
