289template <typename TComponent, typename ... TArgs>
290void Registry::AddComponent(Entity entity, TArgs&& ... args) {
291 const int componentId = Component<TComponent>::GetId();
292 const int entityId = entity.GetId();
293
294 if (static_cast<long unsigned int>(componentId) >= componentsPools.size()) {
295 componentsPools.resize(componentId + 10, nullptr);
296 }
297
298 if (!componentsPools[componentId]) {
299 std::shared_ptr<Pool<TComponent>> newComponentPool
300 = std::make_shared<Pool<TComponent>>();
301 componentsPools[componentId] = newComponentPool;
302 }
303
304 std::shared_ptr<Pool<TComponent>> componentPool
305 = std::static_pointer_cast<Pool<TComponent>>(componentsPools[componentId]);
306
307 if (entityId >= componentPool->GetSize()) {
308 componentPool->Resize(numEntity + 100);
309 }
310
311 TComponent newComponent(std::forward<TArgs>(args)...);
312 componentPool->Set(entityId, newComponent);
313 entityComponentSignatures[entityId].set(componentId);
314}
