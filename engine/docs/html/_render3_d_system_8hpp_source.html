45 // Helper struct to manage face rendering
46 struct RenderableFace {
47 glm::vec3 v1, v2, v3;
48 std::string materialName;
49 float averageDepth;
50 bool isVisible;
51
52 // Calculate face normal to determine visibility
53 bool calculateVisibility() {
54 // Calculate two edge vectors
55 glm::vec3 edge1 = v2 - v1;
56 glm::vec3 edge2 = v3 - v1;
57
58 // Calculate face normal using cross product
59 glm::vec3 normal = glm::cross(edge1, edge2);
60
61 // Simple view direction (For Backface Culling)
62 glm::vec3 viewDirection(0, 0, -1);
63
64 // Face is visible if normal points towards the camera (dot product > 0)
65 return glm::dot(normal, viewDirection) > 0;
66 }
67
68 // Calculate average depth for sorting
69 void computeAverageDepth() {
70 averageDepth = (v1.z + v2.z + v3.z) / 3.0f;
71 }
72 };
73
74 // Depth sorting function
75 static bool compareFaceDepth(const RenderableFace& a, const RenderableFace& b) {
76 // Sort from back to front (furthest to nearest)
77 return a.averageDepth < b.averageDepth;
78 }

130 {
131 // Iterate through vertices and draw triangles using lines
132 for (size_t i = 0; i < vertex.size(); i += 3) {
133 if (i + 2 < vertex.size()) {
134 // Extract vertices (first element of each pair)
135 glm::vec3 v1 = vertex[i].first;
136 glm::vec3 v2 = vertex[i + 1].first;
137 glm::vec3 v3 = vertex[i + 2].first;

166 void drawModel(SDL_Renderer* renderer,
167 const std::vector<std::pair<glm::vec3, std::string>>& vertex,
168 const AssetManager::ObjAsset& objAsset,
169 const ObjectComponent objectC, const TransformComponent transformC)
170 {
171 std::vector<RenderableFace> renderFaces;
172 for (size_t i = 0; i < vertex.size(); i += 3) {
173 if (i + 2 < vertex.size()) {
174 RenderableFace face;
175 face.v1 = vertex[i].first;
176 face.v2 = vertex[i + 1].first;
177 face.v3 = vertex[i + 2].first;
178 face.computeAverageDepth();
179 face.isVisible = face.calculateVisibility();
180
181 // Store the material name for this face
182 face.materialName = vertex[i].second; // Assuming material is consistent for the face
183
184 renderFaces.push_back(face);
185 }
186 }
187
188 std::sort(renderFaces.begin(), renderFaces.end(), compareFaceDepth);
189
190 int offsetX = transformC.position.x;
191 int offsetY = transformC.position.y;
192 glm::vec3 lightDirection(0.0f, 0.0f, -1.0f);
193
194 auto calculateShading = [](const glm::vec3& normal, const glm::vec3& lightDir) -> float {
195 float intensity = glm::dot(normal, glm::normalize(lightDir));
196 return glm::clamp(intensity, 0.5f, 1.0f);
197 };
198
199 for (const auto& face : renderFaces) {
200 if (!face.isVisible) continue;
201
202 glm::vec3 edge1 = face.v2 - face.v1;
203 glm::vec3 edge2 = face.v3 - face.v1;
204 glm::vec3 normal = glm::normalize(glm::cross(edge1, edge2));
205
206 float shadingIntensity = calculateShading(normal, lightDirection);
207
208 // Use the material name stored in the face
209 glm::vec3 baseColor = getMaterialColor(objAsset, face.materialName);
210
211 // Blend shadow color with material color based on shading intensity
212 glm::vec3 shadowColor = glm::vec3(objectC.sr, objectC.sg, objectC.sb);
213 glm::vec3 shadedColor = baseColor * shadingIntensity + shadowColor * (1.0f - shadingIntensity);
214
215 // Convert the shaded color to 8-bit per channel
216 uint8_t shadedRed = static_cast<uint8_t>(shadedColor.r * 255);
217 uint8_t shadedGreen = static_cast<uint8_t>(shadedColor.g * 255);
218 uint8_t shadedBlue = static_cast<uint8_t>(shadedColor.b * 255);
219
220 SDL_Point points[3];
221 points[0] = { static_cast<int>(face.v1.x + offsetX), static_cast<int>(face.v1.y + offsetY) };
222 points[1] = { static_cast<int>(face.v2.x + offsetX), static_cast<int>(face.v2.y + offsetY) };
223 points[2] = { static_cast<int>(face.v3.x + offsetX), static_cast<int>(face.v3.y + offsetY) };
224
225 filledTrigonRGBA(renderer,
226 points[0].x, points[0].y,
227 points[1].x, points[1].y,
228 points[2].x, points[2].y,
229 shadedRed, shadedGreen, shadedBlue, 255);
230 }
231 }

243 std::vector<std::pair<glm::vec3, std::string>> setupVertexArray(
244 const std::vector<glm::vec3>& vertex,
245 const std::vector<AssetManager::Face>& faces)
246 {
247 // Scale factor for vertices
248 float scale = 40.0f;
249 // Initialize vertex array with material information
250 std::vector<std::pair<glm::vec3, std::string>> vertexArray;
251
252 // Process object's faces
253 for (const auto& face : faces)
254 {
255 // Get vertex positions and scale
256 glm::vec3 vertexPosition1 = vertex[face.vertexIndices[0]];
257 glm::vec3 vertexPosition2 = vertex[face.vertexIndices[1]];
258 glm::vec3 vertexPosition3 = vertex[face.vertexIndices[2]];
259
260 glm::vec3 vertexScaled1 = glm::vec3(vertexPosition1.x * scale, -vertexPosition1.y * scale, vertexPosition1.z * scale);
261 glm::vec3 vertexScaled2 = glm::vec3(vertexPosition2.x * scale, -vertexPosition2.y * scale, vertexPosition2.z * scale);
262 glm::vec3 vertexScaled3 = glm::vec3(vertexPosition3.x * scale, -vertexPosition3.y * scale, vertexPosition3.z * scale);
263
264 // Add scaled vertices with their corresponding material name
265 vertexArray.push_back({ vertexScaled1, face.materialName });
266 vertexArray.push_back({ vertexScaled2, face.materialName });
267 vertexArray.push_back({ vertexScaled3, face.materialName });
268 }
269 return vertexArray;
270 }

282 std::vector<std::pair<glm::vec3, std::string>> rotateModel(
283 const std::vector<std::pair<glm::vec3, std::string>>& baseModel,
284 float angleX,
285 float angleY)
286 {
287 // Create a rotation matrix for the X-axis (vertical rotation)
288 glm::mat4 rotationMatrixX = glm::rotate(glm::mat4(1.0f), angleX, glm::vec3(1.0f, 0.0f, 0.0f));
289 // Create a rotation matrix for the Y-axis (horizontal rotation)
290 glm::mat4 rotationMatrixY = glm::rotate(glm::mat4(1.0f), angleY, glm::vec3(0.0f, 1.0f, 0.0f));
291 // Combine the rotation matrices (Y * X for proper order)
292 glm::mat4 combinedRotationMatrix = rotationMatrixY * rotationMatrixX;
293
294 // Apply the combined rotation matrix to each vertex
295 std::vector<std::pair<glm::vec3, std::string>> rotatedVertexArray;
296 for (const auto& [vertex, materialName] : baseModel) {
297 glm::vec4 rotatedVertex = combinedRotationMatrix * glm::vec4(vertex, 1.0f);
298 rotatedVertexArray.push_back({
299 glm::vec3(rotatedVertex),
300 materialName
301 });
302 }
303 return rotatedVertexArray;
304 }

316 glm::vec3 getMaterialColor(const AssetManager::ObjAsset& objAsset, const std::string& materialName) {
