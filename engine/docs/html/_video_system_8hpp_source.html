106 void PlayVideo(const AssetManager::VideoAsset& videoAsset, SDL_Renderer* renderer,
107 VideoComponent& videoComponent, TransformComponent& transformComponent, SDL_Rect& camera) {
108
109 if (!videoAsset.formatCtx || !videoAsset.codecCtx) {
110 return; // Invalid video asset.
111 }
112
113 AVPacket* packet = av_packet_alloc();
114 AVFrame* frame = av_frame_alloc();
115 AVFrame* yuvFrame = av_frame_alloc();
116
117 // Find the video stream index.
118 int videoStreamIndex = -1;
119 for (unsigned int i = 0; i < videoAsset.formatCtx->nb_streams; i++) {
120 if (videoAsset.formatCtx->streams[i]->codecpar->codec_type == AVMEDIA_TYPE_VIDEO) {
121 videoStreamIndex = i;
122 break;
123 }
124 }
125
126 if (videoStreamIndex == -1) {
127 std::cerr << "[VIDEOSYSTEM] Could not find video stream." << std::endl;
128 av_packet_free(&packet);
129 av_frame_free(&frame);
130 av_frame_free(&yuvFrame);
131 return;
132 }
133
134 // Allocate buffer for YUV frame.
135 uint8_t* yuvBuffer = nullptr;
136 int numBytes = av_image_alloc(
137 yuvFrame->data, yuvFrame->linesize,
138 videoAsset.codecCtx->width, videoAsset.codecCtx->height,
139 AV_PIX_FMT_YUV420P, 32);
140
141 if (numBytes < 0) {
142 std::cerr << "[VIDEOSYSTEM] Could not allocate frame buffer." << std::endl;
143 av_packet_free(&packet);
144 av_frame_free(&frame);
145 av_frame_free(&yuvFrame);
146 return;
147 }
148 yuvBuffer = yuvFrame->data[0];
149
150 // Warm-up phase: Decode and discard frames for the first 5 frames.
151 if (videoComponent.warmupCount < 5) {
152 std::cerr << "[VIDEOSYSTEM] Warm-up phase." << std::endl;
153 while (videoComponent.warmupCount < 5) {
154 if (av_read_frame(videoAsset.formatCtx, packet) >= 0) {
155 if (packet->stream_index == videoStreamIndex) {
156 if (avcodec_send_packet(videoAsset.codecCtx, packet) == 0) {
157 avcodec_receive_frame(videoAsset.codecCtx, frame); // Discard the frame.
158 }
159 }
160 av_packet_unref(packet);
161 }
162 videoComponent.warmupCount++; // Increment the warm-up frame count
163 }
164 }
165
166 // Timing control using deltaTime.
167 double frameDuration = 1.0 / av_q2d(
168 videoAsset.formatCtx->streams[videoStreamIndex]->avg_frame_rate); // Frame duration in seconds.
169 static double accumulatedTime = 0.0;
170
171 accumulatedTime += deltaTime;
172
173 // Decode only if enough time has passed.
174 if (accumulatedTime >= frameDuration) {
175 accumulatedTime -= frameDuration; // Reset time.
176
177 // Read frame from video.
178 if (av_read_frame(videoAsset.formatCtx, packet) >= 0) {
179 if (packet->stream_index == videoStreamIndex) {
180 if (avcodec_send_packet(videoAsset.codecCtx, packet) == 0) {
181 if (avcodec_receive_frame(videoAsset.codecCtx, frame) == 0) {
182 // Free the old frame and allocate a new one
183 av_frame_free(&lastDecodedFrame);
184 lastDecodedFrame = av_frame_clone(frame);
185
186 // Convert frame to YUV for SDL.
187 struct SwsContext* swsCtx = sws_getContext(
188 videoAsset.codecCtx->width, videoAsset.codecCtx->height, videoAsset.codecCtx->pix_fmt,
189 videoAsset.codecCtx->width, videoAsset.codecCtx->height, AV_PIX_FMT_YUV420P,
190 SWS_BILINEAR, nullptr, nullptr, nullptr);
191
192 if (!swsCtx) {
193 std::cerr << "[VIDEOSYSTEM] Could not initialize sws context." << std::endl;
194 av_freep(&yuvBuffer);
195 av_packet_free(&packet);
196 av_frame_free(&frame);
197 av_frame_free(&yuvFrame);
198 return;
199 }
200
201 sws_scale(
202 swsCtx, frame->data, frame->linesize, 0, videoAsset.codecCtx->height,
203 yuvFrame->data, yuvFrame->linesize);
204
205 SDL_UpdateYUVTexture(
206 videoAsset.texture, NULL,
207 yuvFrame->data[0], yuvFrame->linesize[0],
208 yuvFrame->data[1], yuvFrame->linesize[1],
209 yuvFrame->data[2], yuvFrame->linesize[2]
210 );
211
212 sws_freeContext(swsCtx);
213 }
214 }
215 }
216 av_packet_unref(packet); // Free packet.
217 }
218 else {
219 // End of video, rewind.
220 av_seek_frame(videoAsset.formatCtx, videoStreamIndex, 0, AVSEEK_FLAG_BACKWARD);
221 avcodec_flush_buffers(videoAsset.codecCtx); // Clear and reset.
222 }
223 }
224
225 // Render even if no new frames (for synchronization).
226
227 if (transformComponent.cameraFree) {
228 SDL_Rect dstRect = { videoComponent.posX, videoComponent.posY,
229 videoComponent.width, videoComponent.height }; // Get position and dimensions.
230 SDL_RenderCopy(renderer, videoAsset.texture, NULL, &dstRect);
231 } else {
232
233 SDL_Rect dstRect = { videoComponent.posX - camera.x, videoComponent.posY - camera.y,
234 videoComponent.width, videoComponent.height }; // Get position and dimensions.
235 SDL_RenderCopy(renderer, videoAsset.texture, NULL, &dstRect);
236 }
237
238 // Free resources.
239 av_freep(&yuvBuffer);
240 av_packet_free(&packet);
241 av_frame_free(&frame);
242 av_frame_free(&yuvFrame);
243 }
